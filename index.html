C Language Solutions



1. Finding the Oldest Age among Three:
#include stdio.h
int main() { int age1, age2, age3;
printf("Enter three ages: ");
scanf("%d %d %d", &age1, &age2, &age3);
int oldest = age1;
if (age2 > oldest) { oldest = age2;
} if (age3 > oldest) { oldest = age3;
}
printf("The oldest age is: %d\n", oldest);
return 0;
}


2. Celsius to Fahrenheit Conversion:
#include stdio.h
int main() { float celsius, fahrenheit;
printf("Enter temperature in Celsius: ");
scanf("%f", &celsius);
fahrenheit = (celsius * 9/5) + 32;
printf("Temperature in Fahrenheit: %.2f\n", fahrenheit);
return 0; }


3. Swapping Two Numbers: #include stdio.h
int main()
{ int num1, num2, temp;
printf("Enter two numbers: "); scanf("%d %d", &num1, &num2);
temp = num1;
num1 = num2;
num2 = temp;
printf("Swapped numbers: %d %d\n", num1, num2);
return 0;
}


4. Sum of Three Digits:
#include stdio.h
int main() {
int num, sum = 0;
printf("Enter a 3-digit number: ");
scanf("%d", &num);
while (num != 0) {
sum += num % 10;
num /= 10;
}
printf("Sum of the digits: %d\n", sum);
return 0;
}


5. Reversing a Four-Digit Number:
#include
int main() {
int num, reversed = 0, original, remainder;
printf("Enter a 4-digit number: ");
scanf("%d", &num);
original = num;
while (num != 0) {
remainder = num % 10;
reversed = reversed * 10 + remainder;
num /= 10;
}
printf("Reversed number: %d\n", reversed);
if (original == reversed) {
printf("The reverse is the same as the original.\n");
} else { printf("The reverse is not the same as the original.\n");
}
return 0;
}


Java Language Solutions


1. Finding the Oldest Age among Three:
import java.util.Scanner;
public class Main {
public static void main(String[] args) {
Scanner scanner = new Scanner(System.in);
System.out.println("Enter three ages: ");
int age1 = scanner.nextInt();
int age2 = scanner.nextInt();
int age3 = scanner.nextInt();
int oldest = Math.max(age1, Math.max(age2, age3));
System.out.println("The oldest age is: " + oldest);
}
}

sum_of_digits = sum(map(int, str(num)))
print("Sum of the digits:", sum_of_digits)

reversed = int(str(num)[::-1])

print("The reverse is the same as the original.")
else:
print("The reverse is not the same as the original.")
